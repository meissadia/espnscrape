def printTable(args, row_width=15, title='')
	spc = row_width
	puts title
	args.each do |row|
		row.each do |td|
			print td
			print " "*(spc - td.to_s.length) if ((spc - td.to_s.length) > 0)
		end
		puts
	end
end
