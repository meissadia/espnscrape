def initialize(espn_player_id)
  espn_player_id = espn_player_id.to_s
  unless (espn_player_id.empty?)
    url = playerUrl + espn_player_id
    doc = Nokogiri::HTML(open(url))
    if doc.nil?
      puts "URL Unreachable: " + url
      exit 1
    end
  end

  readInfo(doc)
end
