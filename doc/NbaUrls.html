def formatTeamUrl(team_id, url)
	team_id = team_id.downcase
	case team_id
	when 'uta'
		team_id = 'utah'
	when 'nop'
		team_id = 'no'
	when 'sas'
		team_id = 'sa'
	when 'was'
		team_id = 'wsh'
	when 'pho'
		team_id = 'phx'
	when 'gsw'
		team_id = 'gs'
	when 'nyk'
		team_id = 'ny'
	end
	return url % [team_id]
end

def getTid(team_name)
	abbr_tmp = team_name.split
	abbr = ""
	if abbr_tmp.size > 2
		abbr_tmp.each do |x|
			abbr << x[0]
		end
		abbr.upcase!
	else
		abbr = abbr_tmp[0][0..2].upcase
	end

	case abbr
	when 'OCT'
		abbr = 'OKC'
	when 'PTB'
		abbr = 'POR'
	when 'BRO'
		abbr = 'BKN'
	end

	return abbr
end
